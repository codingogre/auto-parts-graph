/* All Queries
  1.  GET all Vehicle Lines Related to one or more Parts
  2.  GET all Parts supplied by Tier1(direct supplier)
  3.  GET all T1 supplier for the selected Tier`n` (Indirect suppliers)
  4.  Based on Geo Spatial points GET all T1, T2, T3  Tn suppliers
     *   GET all T1 suppliers based on #4 T2, T3, Tn
     *   GET all Parts supplied by #4 and #4.a T1 Suppliers
  5.  Based on Geo Spatial points GET all Plants(Assembly and Receiving Plants)
     *   GET all T1 suppliers based on #5
     *   GET all Parts  used by #5 Plants
     *   GET all T1 suppliers based on #5

  6.  GET filters for UI
  7.  GET counts for #1 - #5
*/


#1 GET all Vehicle Lines Related to one or more Parts
GET /ford-parts-graph/_search
{
  "query": {
    "bool": {
      "filter": {
        "terms": {
          "partId": ["P001", "P002"]
        }
      }
    }
  },
  "fields": [
    "vehicleLines.vehicleLineName"
  ],
  "_source": false
}

#2 GET all Parts supplied by Tier1(direct supplier)
GET /ford-parts-graph/_search
{
  "query": {
    "nested": {
      "path": "partRoutes",
      "query": {
        "nested": {
          "path": "partRoutes.directSuppliers",
          "query": {
            "bool": {
              "filter": [
                { "term": { "partRoutes.directSuppliers.tier": "T1" } },
                { "term": { "partRoutes.directSuppliers.supplierName": "Supplier DS-0-1" } }
              ]
            }
          }
        }
      }
    }
  },
  "collapse": {
    "field": "partId"         
  },
  "fields": [
    "partId"
  ],
  "_source": false
}
